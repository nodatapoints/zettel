	\[ \lim_{\substack{x \rightarrow x_0 \\ x \neq x_0}} f'(x) \coloneqq \alpha \]
	so ist $f$ in $x_0$ diff'bar mit $f'(x_0) = \alpha$.
\end{theorem}

	\lim_{\varepsilon \rightarrow 0} \frac{f(x_0+\varepsilon)-f(x_0)}{\varepsilon} &\overset{\textsc{L'Hospital}}{=}
\end{align*}
\end{proof}
